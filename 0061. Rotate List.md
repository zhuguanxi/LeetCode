## Problem Description
Given the head of a linked list, rotate the list to the right by k places.

### WAY 1:
```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* rotateRight(ListNode* head, int k) {
        if (head == nullptr)
            return head;

        int len = 0;
        ListNode* cur = head;
        while (cur){
            len++;
            cur = cur->next;
        }

        k = k % len;

        ListNode* tail, *ans;
        cur = tail = head;

        if (k == 0)
            return head;

        for (int i = 0; i < k; i++){
            cur = cur->next;
        }

        while(cur->next){
            cur = cur->next;
            tail = tail->next;
        }

        ans = tail->next;
        tail->next = nullptr;
        cur->next = head;
        return ans;

    }
};
```
* n：The number of nodes in the list is in the range [0, 500]
* k : 0 <= k <= 2 * 109
* Time Complexity $O(k)$
* Space Complexity $O(1)$
