## Problem Description
Given a 0-indexed integer array nums of size n and two integers lower and upper, return the number of fair pairs.

A pair (i, j) is fair if:

* 0 <= i < j < n, and
* lower <= nums[i] + nums[j] <= upper

### WAY 1:
* 使用類似雙指針， 但會TLE
```
class Solution {
public:
    long long countFairPairs(vector<int>& nums, int lower, int upper) {
        sort(nums.begin(), nums.end());

        int ans = 0;
        for (int i=0; i<nums.size(); i++)
        {
            int left = i;
            int right = nums.size() - 1;

            while (left < right)
            {
                int sum = nums[left] + nums[right];
                if (sum <= upper && sum >= lower)
                {
                    ans ++;
                    right --;
                }  
                else if (sum < lower)
                    break;
                else if (sum > upper)
                    right --;
            }
        }
        

        return ans;
    }
};
```
* N：nums 的長度 (1 <= nums.length <= 105)
* Time Complexity $O(N^2)$
* Space Complexity $O(1)$

### WAY 2:
* Sorting + Binary Search
```
class Solution {
public:
    long long countFairPairs(vector<int>& nums, int lower, int upper) {
        sort(nums.begin(), nums.end());
        long long int ans = 0;

        for (int i=0; i<nums.size()-1; i++)
        {
            int min_low = lower - nums[i];
            int max_upp = upper - nums[i];

            auto high = upper_bound(nums.begin() + i + 1, nums.end(), max_upp);
            auto low = lower_bound(nums.begin() + i + 1, nums.end(), min_low);

            ans += (high - low);
        }
        return ans;
    }
};
```
* Time Complexity $O(NlogN)$
* Space Complexity $O(1)$

### WAY 3:
* Sorting + Two Pointers (最優解)
* sort 完只需要花 N 的時間
```
class Solution {
public:
    long long countPairs(vector<int>& nums, long long target) 
    {
        int left = 0;
        int right = nums.size() - 1;
        long long count = 0;

        while (left < right) 
        {
            if (nums[left] + nums[right] <= target) 
            {
                count += right - left;
                left++;
            } 
            else 
                right--;
        }

        return count;
    }

    long long countFairPairs(vector<int>& nums, int lower, int upper) {
        sort(nums.begin(), nums.end());
        return countPairs(nums, upper) - countPairs(nums, lower - 1);
    }
};
```
* Time Complexity $O(NlogN)$
* Space Complexity $O(1)$
