## Problem Description
You are given an m x n matrix grid of positive integers. Your task is to determine if it is possible to make either one horizontal or one vertical cut on the grid such that:

* Each of the two resulting sections formed by the cut is non-empty.
* The sum of the elements in both sections is equal.
Return true if such a partition exists; otherwise return false.

### WAY 1:
* 算出全部的行列總和，並每個例子都算過
```
class Solution {
public:
    bool canPartitionGrid(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();   // n:row m:col

        vector<long long int> row_sum(n);
        vector<long long int> col_sum(m);

        for (int i=0; i<n; i++)
        {
            long long int sum = 0;
            for (int j=0; j<m; j++)
                sum += grid[i][j];
            row_sum[i] = sum;
        }

        for (int i=0; i<m; i++)
        {
            long long int sum = 0;
            for (int j=0; j<n; j++)
                sum += grid[j][i];
            col_sum[i] = sum;
        }

        long long int total = 0;
        for (auto x : row_sum) total += x;

        long long prefix = 0;
        for (int i = 0; i < n - 1; i++)
        {
            prefix += row_sum[i];
            if (prefix == total - prefix)
                return true;
        }

        total = 0;
        for (auto x : col_sum) total += x;

        prefix = 0;
        for (int i = 0; i < m - 1; i++)
        {
            prefix += col_sum[i];
            if (prefix == total - prefix)
                return true;
        }

        return false;
    }
};
```
* m：nums 的長度 (1 <= m == grid.length <= 105)
* n：nums 的長度 (1 <= n == grid[i].length <= 105)
* Time Complexity $O(nm + n + m) = O(nm)$
* Space Complexity $O(n+m)$

### WAY 2:
* 直接算出全部總和並且直接找/2
```
class Solution {
public:
    bool canPartitionGrid(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();   // n:row m:col

        long long total = 0;
        for (auto& row:grid)
            for (auto& val:row)
                total += val;

        if (total % 2 == 1)
            return false;

        long long int target = total / 2;

        long long int cur = 0;
        for (int i=0; i<n-1; i++)
        {
            for (int j=0; j<m; j++)
                cur += grid[i][j];
            if (cur == target) return true;
        }

        cur = 0;
        for (int i=0; i<m-1; i++)
        {
            for (int j=0; j<n; j++)
                cur += grid[j][i];
            if (cur == target) return true;
        }

        return false;
    }
};
```
* Time Complexity $O(nm + nm + nm) = O(nm)$
* Space Complexity $O(1)$
