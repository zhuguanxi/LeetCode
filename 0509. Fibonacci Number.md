### WAY 1:
```
class Solution {
public:
    int fib(int n) {
        vector<int> Fib_arr(n+1, 0);

        if (n == 0)
            return 0;
        else if (n == 1)
            return 1;

        Fib_arr[1] = 1;
        for (int i=2; i<=n; i++)
        {
            Fib_arr[i] = Fib_arr[i-1] + Fib_arr[i-2];
        }

        return Fib_arr[n];
    }
};
```
