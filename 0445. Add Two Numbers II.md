### WAY 1:
* 先反轉 List再加法最後再反轉 answer，但使用 stack 更方便(WAY 2)，
```
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode* new_l1 = reverseList(l1);
        ListNode* new_l2 = reverseList(l2);

        ListNode* dummy = new ListNode(0);
        ListNode* cur = dummy;

        int carry = 0;

        while (new_l1 || new_l2 || carry) 
        {
            int v1 = new_l1 ? new_l1->val : 0;
            int v2 = new_l2 ? new_l2->val : 0;

            int sum = v1 + v2 + carry;

            carry = sum / 10;

            cur->next = new ListNode(sum % 10);
            cur = cur->next;

            if (new_l1) 
                new_l1 = new_l1->next;
            if (new_l2) 
                new_l2 = new_l2->next;
        }

        return reverseList(dummy->next);
    }

    ListNode* reverseList(ListNode* head) {
        ListNode* pre = nullptr;

        while (head != nullptr) {
            ListNode* next = head->next;
            head->next = pre;
            pre = head;
            head = next;
        }

        return pre;
    }
};
```
* N：List 的長度 (The number of nodes in each linked list is in the range [1, 100])
* Time Complexity $O(N)$
* Space Complexity $O(N)$  

### WAY 2:
* 因為題目最高位在前 (MSD first)，所以使用 stack，還不用反轉 List
* while 條件可能包括(list1、list2、最後可能還有 carry)
```
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {

        stack<int> s1;
        stack<int> s2;

        while (l1) 
        {
            s1.push(l1->val);
            l1 = l1->next;
        }

        while (l2)
        {
            s2.push(l2->val);
            l2 = l2->next;
        }

        int carry = 0;
        ListNode* head = nullptr;

        while (!s1.empty() || !s2.empty() || carry) {

            int v1 = 0;
            int v2 = 0;

            if (!s1.empty()) 
            {
                v1 = s1.top();
                s1.pop();
            }

            if (!s2.empty()) 
            {
                v2 = s2.top();
                s2.pop();
            }

            int sum = v1 + v2 + carry;

            carry = sum / 10;

            ListNode* node = new ListNode(sum % 10, head);
            head = node;
        }

        return head;
    }
};
```
* Time Complexity $O(N)$
* Space Complexity $O(N)$  // stack
