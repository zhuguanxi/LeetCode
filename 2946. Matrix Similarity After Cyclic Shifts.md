## Problem Description
You are given an m x n integer matrix mat and an integer k. The matrix rows are 0-indexed.

The following proccess happens k times:
* Even-indexed rows (0, 2, 4, ...) are cyclically shifted to the left.
* Odd-indexed rows (1, 3, 5, ...) are cyclically shifted to the right.

### WAY 1:
* 方法很爛不要學
```
class Solution {
public:
    bool areSimilar(vector<vector<int>>& mat, int k) {
        int n = mat[0].size(), m = mat.size();
        vector<vector<int>> ans(m, vector<int>(n));
        int shift = k % n;
        if (shift == 0) return true;

        for (int t=0; t<shift; t++)
        {
            for (int i=0; i<m; i++)
            {
                int tmp = ans[i][0];
                for (int j=0; j<n; j++)
                {
                    if (j == n-1 && t == 0)
                        ans[i][j] = mat[i][0];
                    else if (j == n-1 && t != 0)
                        ans[i][j] = tmp;
                    else if (j != n-1 && t==0)
                        ans[i][j] = mat[i][j+1];
                    else if (j != n-1 && t!=0)
                        ans[i][j] = ans[i][j+1];
                    cout << ans[i][j];
                }
                cout <<endl;
            }
        }
        return mat == ans ? true : false;
    }
};
```
* n：1 <= mat[i].length <= 25
* m：1 <= mat.length <= 25
* k：1 <= k <= 50
* Time Complexity $O(knm)$
* Space Complexity $O(nm)$

### WAY 2:
* 最優解
```
class Solution {
public:
    bool areSimilar(vector<vector<int>>& mat, int k) {
        int n = mat[0].size(), m = mat.size();

        int shift = k % n;
        if (shift == 0) return true;

        for (int i=0; i<m; i++)
        {
            for (int j=0; j<n; j++)
            {
                if (i % 2 == 0)
                {
                    if (mat[i][j] != mat[i][(j + shift) % n])
                        return false;
                }
                else
                {
                    if (mat[i][j] != mat[i][(j - shift + n) % n])
                        return false;
                }
                    
                        
            }
        }
        return true;
    }
};
```
* Time Complexity $O(nm)$
* Space Complexity $O(1)$
